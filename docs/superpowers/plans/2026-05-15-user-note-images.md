# User Note Images Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add multi-image support to `我的备注`, including paste-to-upload, local-file persistence in the backend, editable note/image combinations in knowledge detail, and thumbnail/lightbox display in review cards.

**Architecture:** Extend `NoteUserNote` to store `content` plus an `images` string-array of served relative URLs. Keep upload and file-cleanup logic inside the backend notes module, expose the upload directory via `ServeStaticModule`, and add a small shared frontend image gallery/lightbox so `KnowledgeDetail` and `ReviewCards` render the same image behavior with different thumbnail sizes.

**Tech Stack:** NestJS 10, Prisma/PostgreSQL, `@nestjs/platform-express`, `@nestjs/serve-static`, React 19, Zustand, Framer Motion, Vitest, Testing Library, Jest/Supertest, `multipart/form-data`

```ts
// backend/src/notes/notes.service.ts
async listUserNotes(noteId: string) {
  await this.ensureActive(noteId)
  const items = await this.prisma.noteUserNote.findMany({
    where: { noteId, deletedAt: null },
    orderBy: { createdAt: 'desc' },
  })
  return { items }
}

async createUserNote(noteId: string, dto: CreateUserNoteDto, files: Express.Multer.File[]) {
  await this.ensureActive(noteId)
  const content = (dto.content ?? '').trim()
  const images = await this.noteUserImageStorage.saveMany(files)
  if (!content && images.length === 0) {
    await this.noteUserImageStorage.removeMany(images)
    throw new BadRequestException('备注内容和图片不能同时为空')
  }

  try {
    return await this.prisma.noteUserNote.create({
      data: { noteId, content, images },
    })
  } catch (error) {
    await this.noteUserImageStorage.removeMany(images)
    throw error
  }
}

async updateUserNote(noteId: string, userNoteId: string, dto: UpdateUserNoteDto, files: Express.Multer.File[]) {
  await this.ensureActive(noteId)
  const existing = await this.prisma.noteUserNote.findFirst({
    where: { id: userNoteId, noteId, deletedAt: null },
  })
  if (!existing) throw new NotFoundException('User note not found')

  const keepImages = this.parseKeepImages(dto.keepImages, existing.images)
  const addedImages = await this.noteUserImageStorage.saveMany(files)
  const nextImages = [...keepImages, ...addedImages]
  const content = (dto.content ?? existing.content ?? '').trim()
  if (!content && nextImages.length === 0) {
    await this.noteUserImageStorage.removeMany(addedImages)
    throw new BadRequestException('备注内容和图片不能同时为空')
  }

  try {
    const updated = await this.prisma.noteUserNote.update({
      where: { id: userNoteId },
      data: { content, images: nextImages },
    })
    await this.noteUserImageStorage.removeMany(existing.images.filter((path) => !keepImages.includes(path)))
    return updated
  } catch (error) {
    await this.noteUserImageStorage.removeMany(addedImages)
    throw error
  }
}
```

```ts
// backend/src/notes/note-user-image.storage.ts
import { Injectable, BadRequestException } from '@nestjs/common'
import { mkdir, rm, writeFile } from 'node:fs/promises'
import { extname, join } from 'node:path'
import { randomUUID } from 'node:crypto'

const MIME_TO_EXT: Record<string, string> = {
  'image/png': '.png',
  'image/jpeg': '.jpg',
  'image/webp': '.webp',
  'image/gif': '.gif',
}

@Injectable()
export class NoteUserImageStorage {
  private readonly root = process.env.NOTE_USER_IMAGE_ROOT ?? join(process.cwd(), 'uploads', 'note-user-images')

  async saveMany(files: Express.Multer.File[]) {
    const valid = files.filter((file) => file?.buffer?.length)
    for (const file of valid) {
      if (!(file.mimetype in MIME_TO_EXT)) {
        throw new BadRequestException(`不支持的图片类型: ${file.mimetype}`)
      }
    }

    const now = new Date()
    const year = String(now.getFullYear())
    const month = String(now.getMonth() + 1).padStart(2, '0')
    const dir = join(this.root, year, month)
    await mkdir(dir, { recursive: true })

    const results: string[] = []
    for (const file of valid) {
      const ext = MIME_TO_EXT[file.mimetype] ?? extname(file.originalname) ?? '.bin'
      const filename = `${randomUUID()}${ext}`
      await writeFile(join(dir, filename), file.buffer)
      results.push(`/note-user-images/${year}/${month}/${filename}`)
    }
    return results
  }

  async removeMany(paths: string[]) {
    await Promise.all(paths.map(async (publicPath) => {
      const relative = publicPath.replace(/^\/note-user-images\//, '')
      if (!relative) return
      await rm(join(this.root, relative), { force: true })
    }))
  }
}

// backend/src/notes/notes.module.ts
@Module({
  controllers: [NotesController],
  providers: [NotesService, NoteUserImageStorage],
})
export class NotesModule {}
```

```ts
export function clipboardImageFiles(event: Pick<ClipboardEvent, 'clipboardData'>): File[] {
  const items = Array.from(event.clipboardData?.items ?? [])
  return items

    .map((item) => item.getAsFile())
```

```tsx
// frontend/src/components/ui/UserNoteImageStrip.tsx
export function UserNoteImageStrip({
  images,
  thumbnailClassName,
  maxVisible,
}: {
  images: string[]
  thumbnailClassName: string
  maxVisible?: number
}) {
  const [openIndex, setOpenIndex] = useState<number | null>(null)
  const visible = typeof maxVisible === 'number' ? images.slice(0, maxVisible) : images
  const hiddenCount = images.length - visible.length

  return (
    <>
      <div className="flex flex-wrap gap-2">
        {visible.map((image, index) => (
          <button
            key={`${image}-${index}`}
            type="button"
            aria-label="查看备注图片"
            className={thumbnailClassName}
            onClick={() => setOpenIndex(index)}
          >
            <img src={image} alt="" className="w-full h-full object-cover rounded-md" />
          </button>
        ))}
        {hiddenCount > 0 && <div className="text-xs text-text-dim self-center">+{hiddenCount}</div>}
      </div>
      {openIndex !== null && (
        <UserNoteImageLightbox images={images} startIndex={openIndex} onClose={() => setOpenIndex(null)} />
      )}
    </>
  )
}
```

```tsx
import type { UserNote } from '../../types/userNote'

export function UserNoteCard({

  thumbnailClassName,
  maxVisibleImages,
  actions,

  thumbnailClassName: string
  maxVisibleImages?: number
  actions?: React.ReactNode

      <div className="flex-1 min-w-0">
        {hasText && <p className="text-[13px] text-text-muted leading-relaxed">{note.content}</p>}
        {hasImages && (
          <div className={hasText ? 'mt-2' : ''}>
            <UserNoteImageStrip
              images={note.images}
              thumbnailClassName={thumbnailClassName}
              maxVisible={maxVisibleImages}
            />
          </div>
        )}
      </div>
      {actions}
    </div>
  )
}
```
